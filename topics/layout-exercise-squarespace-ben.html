/* layout-exercise.css - Squarespace Benzeri Layout */ /* RESET - Tüm varsayılan margin/padding'leri sıfırla */ * { margin: 0; padding: 0; box-sizing: border-box; } /* Body margin'ini kaldır */ body { margin: 0; font-family: 'Syne', sans-serif; line-height: 1.6; } /* ========== NAVBAR ========== */ /* NAVBAR ÖZELLİKLERİ: - Yükseklik: 75px - Arka plan: royalblue - İçerik ortalanmış - Yatay menü */ .navbar { height: 75px; background-color: royalblue; display: flex; align-items: center; justify-content: space-between; padding: 0 100px; /* NOT: display: flex ile yan yana yerleştirme yapıyoruz! */ } .logo { font-size: 1.5rem; font-weight: bold; color: white; } .nav-links { display: flex; gap: 2rem; list-style: none; } .nav-links a { color: white; text-decoration: none; } /* ========== HEADER (HERO) ========== */ /* HEADER ÖZELLİKLERİ: - Yükseklik: 400px - Arka plan: lightskyblue - İçerik ortalanmış */ .hero { height: 400px; background-color: lightskyblue; display: flex; flex-direction: column; align-items: center; justify-content: center; text-align: center; color: white; } .hero h1 { font-size: 3rem; margin-bottom: 1rem; } .hero p { font-size: 1.2rem; } /* ========== MAIN SECTION ========== */ /* MAIN SECTION ÖZELLİKLERİ: - Dış boşluk: üst/alt 75px, sağ/sol 100px - Yükseklik: 500px - Arka plan: gray */ .main-section { margin: 75px 100px; background-color: gray; padding: 2rem; /* Yükseklik içerik tarafından belirlensin, sabit 500px yerine min-height */ min-height: 500px; } /* ========== İÇERİK KUTULARI ========== */ /* İKİ KUTUYU YAN YANA KOYMAK: - display: flex ile yan yana diziyoruz - gap ile aralarındaki boşluğu ayarlıyoruz - Her kutu: width: 48% (aradaki boşluk için %2 bırakıyoruz) */ .content-wrapper { display: flex; gap: 4%; /* %4 boşluk → her kutu %48 olursa toplam %100 */ margin-bottom: 75px; } .content-box { width: 48%; background-color: tomato; height: 400px; border-radius: 12px; display: flex; align-items: center; justify-content: center; font-size: 1.5rem; font-weight: bold; color: white; } /* Sarı paragraf */ .yellow-paragraph { background-color: khaki; height: 200px; margin-bottom: 75px; border-radius: 12px; display: flex; align-items: center; justify-content: center; font-size: 1.2rem; font-weight: bold; color: #333; } /* ========== FOOTER ========== */ /* FOOTER ÖZELLİKLERİ: - Yükseklik: 100px - Arka plan: black - İçerik ortalanmış */ .footer { height: 100px; background-color: black; display: flex; align-items: center; justify-content: center; color: white; } /* ========== RESPONSIVE ========== */ @media (max-width: 768px) { .navbar { padding: 0 20px; flex-direction: column; height: auto; padding: 1rem; } .main-section { margin: 40px 20px; } .content-wrapper { flex-direction: column; gap: 20px; } .content-box { width: 100%; } .hero h1 { font-size: 2rem; } }
